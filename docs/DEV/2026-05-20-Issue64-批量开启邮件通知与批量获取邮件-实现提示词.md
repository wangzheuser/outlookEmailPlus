```python
def api_batch_notification_toggle() -> Any:
```

**实现逻辑**（严格按照以下步骤）：

```python
# 1. 解析请求体
data = request.get_json(silent=True) or {}
account_ids = data.get("account_ids", [])
enabled = bool(data.get("enabled", False))

# 2. 参数校验（仿照 api_batch_update_status 模式）
if not isinstance(account_ids, list) or not account_ids:
    return build_error_response("ACCOUNT_IDS_REQUIRED", ...)

try:
    parsed_ids = [int(aid) for aid in account_ids]
except (TypeError, ValueError):
    return build_error_response("INVALID_PARAM", ...)

# 3. 去重
deduped_ids = list(dict.fromkeys(parsed_ids))  # 保持顺序去重

# 4. 批量操作
updated_count = 0
failed_count = 0
missing_ids = []

from outlook_web.repositories.accounts import toggle_telegram_push

for aid in deduped_ids:
    success = toggle_telegram_push(aid, enabled)
    if success:
        updated_count += 1
    else:
        failed_count += 1
        missing_ids.append(aid)

# 5. 返回结构化结果
return jsonify({
    "success": True,
    "enabled": enabled,
    "updated_count": updated_count,
    "failed_count": failed_count,
    "missing_ids": missing_ids,
    "message": f"成功{'开启' if enabled else '关闭'} {updated_count} 个账号通知" + 
               (f"，{failed_count} 个失败" if failed_count else ""),
})
```

**注意事项**：
- `toggle_telegram_push()` 内置幂等逻辑（`repositories/accounts.py:668-671`），重复调用不报错
- 不需要单独处理 `claimed`/`frozen` 状态（通知开关与池状态无关）
- 使用 `build_error_response()` 函数返回错误（项目已有工具函数）

```javascript
async function batchNotificationToggle(enabled) {
    if (selectedAccountIds.size === 0) {
        showToast(translateAppTextLocal('请选择要操作通知的账号'), 'error');
        return;
    }
    const ids = Array.from(selectedAccountIds);
    try {
        const response = await fetch('/api/accounts/batch-notification-toggle', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ account_ids: ids, enabled })
        });
        const data = await response.json();
        if (data.success) {
            showToast(data.message || 
                (enabled ? '批量开启通知完成' : '批量关闭通知完成'), 
                'success');
            if (currentGroupId) {
                delete accountsCache[currentGroupId];
                loadAccountsByGroup(currentGroupId, true);
            }
        } else {
            handleApiError(data, '批量操作失败');
        }
    } catch (error) {
        showToast(translateAppTextLocal('操作失败'), 'error');
    }
}
```
